--------about base-----------------------------------------------
git init 
	create a directory
git add	<file>
	add files to commit
git commit -m "message"	
	commit files,parameter -m is commit message
git status	
	directory status
git diff <file>	
	show the edited in the file
git diff HEAD -- <file>
	show file differences in working directory and branch
git reflog 	
	show log,if terminal has been closed
git log -<n>	
	show log,<n> is the numbers of log to show

--------about rollback-----------------------------------------------
git reset --hard HEAD^
		rollback the directory
git reset HEAD <file>
		unstage file
git checkout -- <file>
	woking derectory file back to the last git add or git commit statu

git rm <file> | git rm -r <folder>
	delete file in branch

--------about remote-------------------------------------------------
git remote add origin git@server -name:path/repo-name.git
	relate remote and directory
git pull
git fetch
git push -u origin master
	pull master branch
git clone 
	e.g:git clone git@10.92.34.214:/usr/test/source.git
--------about branch-------------------------------------------------
git branch 
	display current branch
git branch <branch>
	create a new branch
git branch -d <branch>
	delete branch

git checkout <branch>
	switch to branch
git checkout -b <branch>
	create and switch branch

git merge <branch>
	merge branch to current branch(current branch is older)
	maby you have a conflict
